## get stabililization values. Shakiness 1-10, accuracy 1-15.
ffmpeg -i input.mp4 -vf vidstabdetect=shakiness=10:accuracy=15:result=my_stabilization.trf -f null -

ffmpeg -i input.mp4 -vf vidstabtransform=input=my_stabilization.trf:smoothing=30 -c:v libx264 -preset slow -crf 18 -c:a copy output.mp4

## combine videos to compares
ffmpeg -i input.mp4 -i output.mp4 -filter_complex "hstack" comparison.mp4
